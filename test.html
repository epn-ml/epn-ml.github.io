import os
import tarfile
import urllib
DOWNLOAD_ROOT = "https://raw.githubusercontent.com/epn-ml/LMSU-Mercury_boundaries/main/"
DATA_PATH = os.path.join("csv_files")
DATA_URL = DOWNLOAD_ROOT + "csv_files/dummy.txt"

def fetch_data(data_url=DATA_URL, data_path=DATA_PATH):
    os.makedirs(data_path, exist_ok=True)
    tgz_path = os.path.join(data_path, "data.tgz")
    urllib.request.urlretrieve(data_url, tgz_path)
    data_tgz = tarfile.open(tgz_path)
    data_tgz.extractall(path=data_path)
    data_tgz.close()

from ipyleaflet import Map, Heatmap
from random import uniform
import time

def create_random_data(length):
    "Return a list of some random lat/lon/value triples."
    return [[uniform(-80, 80),
         uniform(-180, 180),
         uniform(0, 1000)] for i in range(length)]

m = Map(center=[0, 0], zoom=2)
m
